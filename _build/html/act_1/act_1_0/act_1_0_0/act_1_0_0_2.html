import networkx as nx
import matplotlib.pyplot as plt

# Create an empty directed graph
G = nx.DiGraph()

# Add nodes
nodes = ['1.Confounder', '3.Replication', '4.Environment', '5.Mediator', '6.Outcome', '2.Heredity']
G.add_nodes_from(nodes)

# Add edges to the graph (arrows in the DAG)
edges = [
    ('1.Confounder', '4.Environment'),
    ('2.Heredity', '4.Environment'),
    ('4.Environment', '5.Mediator'),
    ('5.Mediator', '6.Outcome')
]
G.add_edges_from(edges)

# Set figure size
plt.figure(figsize=(10, 6))  # You can adjust the width and height here

# Draw the DAG
pos = {
    '1.Confounder': (0, 1),
    '3.Replication': (0.2, 0.5),
    '4.Environment': (.75, 0.5),
    '2.Heredity': (0.1, 0),
    '5.Mediator': (1.35, 0.5),
    '6.Outcome': (2, 0.5)
}

labels = {edge: '' for edge in G.edges}

# https://matplotlib.org/stable/api/_as_gen/matplotlib.patches.ConnectionStyle.html#matplotlib.patches.ConnectionStyle
nx.draw(G, pos, with_labels=True, node_color='lightblue', font_weight='normal', node_size=12000, font_size=17, arrows=True, connectionstyle='arc3,rad=0.01')
plt.show()

  4. Environment

    • Hypothesis

    • Cognitive frailty predicts delirium in liver transplant

    • Its defined as:

    • Cognitive frailty is defined as the presence of both physical frailty and cognitive impairment without dementia

    • Physical frailty is defined as the presence of three or more of the following five criteria: unintentional weight loss, self-reported exhaustion, weakness, slow walking speed, and low physical activity

    • Cognitive impairment without dementia is defined as a score of 0.5 standard deviations or more below the mean on two or more tests of cognitive function

    • Cognitive frailty is a risk factor for delirium in liver transplant recipients
